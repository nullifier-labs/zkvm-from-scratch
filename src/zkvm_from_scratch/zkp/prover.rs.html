58    pub fn generate_execution_trace(
59        &self,
60        vm_state: &mut VmState,
61        max_steps: usize,
62    ) -> Result<ExecutionTrace, &'static str> {
63        let initial_state = vm_state.clone();
64        let execution_steps = vm_state.run_with_trace(max_steps)?;
65        let final_state = vm_state.clone();
66
67        let mut trace_steps = Vec::new();
68        for (index, exec_step) in execution_steps.iter().enumerate() {
69            let mut memory_accesses = Vec::new();
70
71            // Convert memory reads
72            for (addr, value) in &exec_step.memory_reads {
73                memory_accesses.push(MemoryAccess {
74                    addr: *addr,
75                    value_before: *value,
76                    value_after: *value,
77                    is_write: false,
78                });
79            }
80
81            // Convert memory writes
82            for (addr, old_value, new_value) in &exec_step.memory_writes {
83                memory_accesses.push(MemoryAccess {
84                    addr: *addr,
85                    value_before: *old_value,
86                    value_after: *new_value,
87                    is_write: true,
88                });
89            }
90
91            trace_steps.push(TraceStep {
92                step_index: index,
93                pc_before: exec_step.pc_before,
94                pc_after: exec_step.pc_after,
95                registers_before: exec_step.registers_before,
96                registers_after: exec_step.registers_after,
97                instruction: exec_step.instruction,
98                memory_accesses,
99                intermediate_values: exec_step.intermediate_values.clone(),
100            });
101        }
102
103        Ok(ExecutionTrace {
104            steps: trace_steps,
105            initial_state,
106            final_state,
107        })
108    }
