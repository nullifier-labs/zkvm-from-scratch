11#[derive(Debug, Clone)]
12pub enum Constraint {
13    Equality {
14        left: Expression,
15        right: Expression,
16    },
17    RangeCheck {
18        value: Expression,
19        max_bits: u32,
20    },
21    MemoryConsistency {
22        addr: Expression,
23        value_read: Expression,
24        value_written: Expression,
25    },
26}

28#[derive(Debug, Clone)]
29pub enum Expression {
30    Constant(u32),
31    Variable(String),
32    Add(Box<Expression>, Box<Expression>),
33    Mul(Box<Expression>, Box<Expression>),
34    Sub(Box<Expression>, Box<Expression>),
35}

64    pub fn generate_constraints_for_trace(&mut self, trace: &ExecutionTrace) {
65        // Add standard witness columns
66        self.add_witness_column("step".to_string());
67        self.add_witness_column("pc_before".to_string());
68        self.add_witness_column("pc_after".to_string());
69
70        for i in 0..32 {
71            self.add_witness_column(format!("reg_{i}_before"));
72            self.add_witness_column(format!("reg_{i}_after"));
73        }
74
75        // Add public columns for initial and final state
76        self.add_public_column("initial_pc".to_string());
77        self.add_public_column("final_pc".to_string());
78
79        for (step_idx, step) in trace.steps.iter().enumerate() {
80            self.generate_constraints_for_step(step, step_idx);
81        }
82    }

84    fn generate_constraints_for_step(&mut self, step: &TraceStep, step_idx: usize) {
85        let _step_var = Expression::Variable(format!("step_{step_idx}"));
86        let pc_before = Expression::Variable(format!("pc_before_{step_idx}"));
87        let pc_after = Expression::Variable(format!("pc_after_{step_idx}"));
88
89        // PC progression constraint for most instructions
90        match step.instruction.opcode {
91            Opcode::Branch | Opcode::Jump => {
92                // Branch/Jump instructions handle PC differently
93            }
94            _ => {
95                // Normal instructions: pc_after = pc_before + 4
96                self.add_constraint(Constraint::Equality {
97                    left: pc_after.clone(),
98                    right: Expression::Add(
99                        Box::new(pc_before.clone()),
100                        Box::new(Expression::Constant(4)),
101                    ),
102                });
103            }
104        }
105
106        // Instruction-specific constraints
107        match step.instruction.opcode {
108            Opcode::Add => {
109                let rs1_val = Expression::Variable(format!(
110                    "reg_{}_before_{}",
111                    step.instruction.rs1, step_idx
112                ));
113                let rs2_val = Expression::Variable(format!(
114                    "reg_{}_before_{}",
115                    step.instruction.rs2, step_idx
116                ));
117                let rd_val =
118                    Expression::Variable(format!("reg_{}_after_{}", step.instruction.rd, step_idx));
119
120                // rd = rs1 + rs2
121                self.add_constraint(Constraint::Equality {
122                    left: rd_val,
123                    right: Expression::Add(Box::new(rs1_val), Box::new(rs2_val)),
124                });
125            }
126            Opcode::Sub => {
127                let rs1_val = Expression::Variable(format!(
128                    "reg_{}_before_{}",
129                    step.instruction.rs1, step_idx
130                ));
131                let rs2_val = Expression::Variable(format!(
132                    "reg_{}_before_{}",
133                    step.instruction.rs2, step_idx
134                ));
135                let rd_val =
136                    Expression::Variable(format!("reg_{}_after_{}", step.instruction.rd, step_idx));
137
138                // rd = rs1 - rs2
139                self.add_constraint(Constraint::Equality {
140                    left: rd_val,
141                    right: Expression::Sub(Box::new(rs1_val), Box::new(rs2_val)),
142                });
143            }
144            Opcode::Mul => {
145                let rs1_val = Expression::Variable(format!(
146                    "reg_{}_before_{}",
147                    step.instruction.rs1, step_idx
148                ));
149                let rs2_val = Expression::Variable(format!(
150                    "reg_{}_before_{}",
151                    step.instruction.rs2, step_idx
152                ));
153                let rd_val =
154                    Expression::Variable(format!("reg_{}_after_{}", step.instruction.rd, step_idx));
155
156                // rd = rs1 * rs2
157                self.add_constraint(Constraint::Equality {
158                    left: rd_val,
159                    right: Expression::Mul(Box::new(rs1_val), Box::new(rs2_val)),
160                });
161            }
162            Opcode::Load | Opcode::Store => {
163                // Memory access constraints
164                for mem_access in &step.memory_accesses {
165                    self.add_constraint(Constraint::MemoryConsistency {
166                        addr: Expression::Constant(mem_access.addr),
167                        value_read: Expression::Constant(mem_access.value_before),
168                        value_written: Expression::Constant(mem_access.value_after),
169                    });
170                }
171            }
172            _ => {}
173        }
174
175        // Register 0 is always 0 constraint
176        self.add_constraint(Constraint::Equality {
177            left: Expression::Variable(format!("reg_0_after_{step_idx}")),
178            right: Expression::Constant(0),
179        });
180
181        // Range checks for register values (32-bit)
182        for i in 0..32 {
183            self.add_constraint(Constraint::RangeCheck {
184                value: Expression::Variable(format!("reg_{i}_after_{step_idx}")),
185                max_bits: 32,
186            });
187        }
188    }

190    #[allow(clippy::only_used_in_recursion)]
191    pub fn evaluate_expression(
192        &self,
193        expr: &Expression,
194        witness: &std::collections::HashMap<String, u32>,
195    ) -> Option<u32> {
196        match expr {
197            Expression::Constant(val) => Some(*val),
198            Expression::Variable(name) => witness.get(name).copied(),
199            Expression::Add(left, right) => {
200                let left_val = self.evaluate_expression(left, witness)?;
201                let right_val = self.evaluate_expression(right, witness)?;
202                Some(left_val.wrapping_add(right_val))
203            }
204            Expression::Mul(left, right) => {
205                let left_val = self.evaluate_expression(left, witness)?;
206                let right_val = self.evaluate_expression(right, witness)?;
207                Some(left_val.wrapping_mul(right_val))
208            }
209            Expression::Sub(left, right) => {
210                let left_val = self.evaluate_expression(left, witness)?;
211                let right_val = self.evaluate_expression(right, witness)?;
212                Some(left_val.wrapping_sub(right_val))
213            }
214        }
215    }

217    pub fn verify_constraints(&self, witness: &std::collections::HashMap<String, u32>) -> bool {
218        for constraint in &self.constraints {
219            if !self.verify_constraint(constraint, witness) {
220                return false;
221            }
222        }
223        true
224    }
225
226    pub fn verify_constraint(
227        &self,
228        constraint: &Constraint,
229        witness: &std::collections::HashMap<String, u32>,
230    ) -> bool {
231        match constraint {
232            Constraint::Equality { left, right } => {
233                let left_val = self.evaluate_expression(left, witness);
234                let right_val = self.evaluate_expression(right, witness);
235                left_val == right_val && left_val.is_some()
236            }
237            Constraint::RangeCheck { value, max_bits } => {
238                if let Some(val) = self.evaluate_expression(value, witness) {
239                    if *max_bits >= 32 {
240                        true // u32 values are always within 32-bit range
241                    } else {
242                        val < (1u32 << max_bits)
243                    }
244                } else {
245                    false
246                }
247            }
248            Constraint::MemoryConsistency { .. } => {
249                // For now, assume memory consistency is always satisfied
250                // In a real implementation, this would check memory operation ordering
251                true
252            }
253        }
254    }
